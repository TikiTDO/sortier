export class Greeter {
  static nextGreeterId: number;
  baseSalary: number;
  description: string;
  greeting: string;
  id: number;
  name: string;
  title: string;

  constructor(message?: string) {
    this.id = Greeter.nextGreeterId;
    Greeter.nextGreeterId++;
    this.greeting = message || "Hi there!";
  }

  greet() {
    return "Hello, " + this.greeting;
  }

  murmour() {
    return "Hello, " + this.greeting;
  }
}

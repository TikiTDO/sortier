/* Switch case example */
switch (1) {
  // This is the first context group
  case 4:
    break;
  case 3: {
    alert(3);
    // Leads into 2
  }
  case 2:
    break;
    
  // This is the second context group
  case 7:
    break;
  case 4:
  case 5:
    break;
}
